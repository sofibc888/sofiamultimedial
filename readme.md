<!DOCTYPE html>
<html>

<head>
    <meta charset="UTF-8">
    <title>Sofia multimedial</title>

    <style>
        /* estilos general */
        body {
            background-color: #DFF2FE;
            color: #1D293D;
            margin: 0;
            height: 100vh;
            font-family: "Times New Roman", serif;
        }

        /* estilos Contenedor principal, (titulos y subtitulos) centrado */
        .contenedor {
            text-align: center;
            margin-top: 40px;
            
        }

        /* Título grande */
        h1 {
            font-size: 70px;
            margin-bottom: 10px;
            
        }

        /* Nombre debajo del título */
        h2 {
            font-size: 30px;
            margin-top: 0;
        }

        /* hace que la Imagen este centrada */
        img {
            display: block;
            margin: 40px auto;
        }

        /* hace que los Links esten en la esquina */
        .nav {
            position: absolute;
            top: 20px;
            right: 20px;
        }
        /* color de los  Links*/
        .nav a {
            display: block;
            font-size: 16px;
            text-decoration: none;
            color: #1D293D;
            margin-bottom: 5px;
        }

        .nav a:hover {
            text-decoration: underline;
        }

    </style>
</head>

<body>

    <!-- Links en la esquina -->
    <div class="nav">
        <a href="Obras.html">Obras</a>
        <a href="Contactos.html">Contactos</a>
    </div>
