<!DOCTYPE html>
<!-- Indica que el documento usa un lenguaje HTML5 que se utiliza para crear paginas web-->

<html lang="es">
<!-- Idioma del contenido -->

<head>
    <meta charset="UTF-8">
    <!-- hace que el codigo Permita mostrar correctamente tildes y caracteres especiales como: !@#$-->

    <title>SOFIA BECERRA</title>
    <!-- Título que aparece en la pestaña del navegador -->

      <!-- En el stlye se guardan todos los estilos -->
    <style>  
       
        /* En el Body se guardan los Estilos generales de la página */
       body {
           background-image: url(img/corderocomplete.jpg); /* Imagen del fondo */
           background-size: cover; /* Hace que la imagen cubra toda la pantalla */
           background-position: center 30%; /* Centra la imagen */
           background-repeat: no-repeat; /* Evita que se repita */
           min-height: 40vh; 
           margin: 0;
           font-family: Times New Roman , Times, serif;
       }

        /* Contenedor de la barra de navegación */
        .navbar {
            display: flex; /* Lo convierte en un  contenedor flexible, lo que hace que se ordene mas facil, es decir que los elementos internos se ordenan en fila */
            justify-content: space-between; /* Hace que el contenido se justifique de lka derecha y la izquierda */
            align-items: center; /* hace que los items o contenido se centren*/
            padding: 20px 40px; /* es la separacion entre los elementos*/
            background-color: transparent; /* hace que el Fondo sea  transparente */
        }
        /* Estilo del contenedor logo (nombre) */
        .logo {
            color: white;
            font-size: 40px;  
            letter-spacing: 4px; /* Espaciado entre letras (como la imagen) */
        }

        /* Contenedor de links */
        .menu a {
            color: white;
            text-decoration: none;/* Elimina la decoracion por defecto, por ejemplo los link suelen ser azules y subrayados  */
            margin-left: 20px;padding: 8px 16px; /* es la separacion entre los elementos*/
            border:  2px solid white; /* estilo botón */
        }
        /* Hover es una funcion para el boton cuando pasas por encima el mouse */
        .menu a:hover {
            background-color: white;
            color: black;
        }
        
    </style>
</head>

<body>

    <!-- Barra de navegación -->

       JAVASCRIPT (lógica del carrusel)

    <script>
        // ================================================
        // DECLARACIÓN DE VARIABLES
        // ================================================
        
        // Selecciona el contenedor que tiene todas las imágenes en fila
        const imagenes = document.getElementById('carruselImagenes');
        
        // Calcula cuántas imágenes hay (6 en total)
        const total = document.querySelectorAll('.imagenes img').length;
        
        // Variable que guarda en qué imagen estamos (empieza en 0 = primera)
        let indice = 0;

        // Selecciona los botones de las flechas
        const anterior = document.getElementById('btnAnterior');
        const siguiente = document.getElementById('btnSiguiente');

        // ================================================
        // FUNCIÓN: Actualizar la posición del carrusel
        // ================================================
        function moverCarrusel() {
            // Mueve el carrusel: si índice=0 → 0%, índice=1 → -100%, índice=2 → -200%
            imagenes.style.transform = `translateX(-${indice * 100}%)`;
            
            // Actualizar los puntos (para que se vea cuál está activo)
            const puntos = document.querySelectorAll('.punto');
            puntos.forEach((punto, i) => {
                if (i === indice) {
                    punto.classList.add('activo'); // Punto activo se pone oscuro
                } else {
                    punto.classList.remove('activo'); // Puntos inactivos se ponen grises
                }
            });
        }

        // ================================================
        // FUNCIÓN: Ir a la SIGUIENTE imagen
        // ================================================
        function siguienteImagen() {
            if (indice === total - 1) {
                indice = 0; // Si es la última imagen, vuelve a la primera (efecto circular)
            } else {
                indice++; // Sino, avanza a la siguiente
            }
            moverCarrusel();
        }

        // ================================================
        // FUNCIÓN: Ir a la ANTERIOR imagen
        // ================================================
        function anteriorImagen() {
            if (indice === 0) {
                indice = total - 1; // Si es la primera imagen, va a la última (efecto circular)
            } else {
                indice--; // Sino, retrocede a la anterior
            }
            moverCarrusel();
        }

        // ================================================
        // FUNCIÓN: Ir a una imagen ESPECÍFICA (al hacer clic en un punto)
        // ================================================
        function irAImagen(posicion) {
            indice = posicion;
            moverCarrusel();
        }

        // ================================================
        // FUNCIÓN: Crear los puntos indicadores automáticamente
        // ================================================
        function crearPuntos() {
            const contenedorPuntos = document.getElementById('puntos');
            contenedorPuntos.innerHTML = ''; // Limpiar el contenedor por si acaso
            
            // Crear un punto por cada imagen
            for (let i = 0; i < total; i++) {
                const punto = document.createElement('div');
                punto.classList.add('punto');
                if (i === 0) punto.classList.add('activo'); // El primer punto empieza activo
                punto.addEventListener('click', () => irAImagen(i)); // Al hacer clic, va a esa imagen
                contenedorPuntos.appendChild(punto);
            }
        }

        // ================================================
        // CONECTAR LOS BOTONES A LAS FUNCIONES
        // ================================================
        siguiente.addEventListener('click', siguienteImagen);
        anterior.addEventListener('click', anteriorImagen);

        // ================================================
        // NAVEGACIÓN CON TECLADO (flechas ← y →)
        // ================================================
        document.addEventListener('keydown', (e) => {
            if (e.key === 'ArrowRight') {
                siguienteImagen(); // Flecha derecha: imagen siguiente
            }
            if (e.key === 'ArrowLeft') {
                anteriorImagen(); // Flecha izquierda: imagen anterior
            }
        });

        // ================================================
        // INICIAR TODO CUANDO CARGUE LA PÁGINA
        // ================================================
        crearPuntos();
    </script>
